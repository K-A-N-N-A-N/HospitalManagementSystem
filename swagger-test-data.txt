# Get All Doctors

# Get Doctor by ID
GET /doctors/1

# Create Doctor
POST /doctors
Content-Type: application/json

{
  "name": "Dr. John Smith",
  "specialization": "Cardiology",
  "contactInfo": "dr.john@example.com",
  // Only if applicable
  "photoPath": "/images/doctors/john-smith.jpg" 
}

# Update Doctor
PUT /doctors/1
Content-Type: application/json

{
  "name": "Dr. John Smith",
  "specialization": "Cardiovascular Surgery",
  "contactInfo": "dr.john.smith@example.com",
  "photoPath": "/images/doctors/john-smith-updated.jpg"
}

# Delete Doctor
DELETE /doctors/1

# ============================================================
# Doctor Slot API
# ============================================================

# Generate Default Slots for a Doctor
POST /api/doctor-slots/generate
Content-Type: application/json

{
  "doctorId": 1,
  "date": "2025-11-15"
}

# Get Scheduled Slots for a Doctor
POST /api/doctor-slots/scheduledSlots
Content-Type: application/json

{
  "doctorId": 1,
  "date": "2025-11-15"
}

# Get Available Slots for a Doctor
POST /api/doctor-slots/available
Content-Type: application/json

{
  "doctorId": 1,
  "date": "2025-11-15"
}

# Update Slot Availability
POST /api/doctor-slots/updateSlot
Content-Type: application/json

{
  "doctorId": 1,
  "date": "2025-11-15",
  "startTime": "09:00:00",
  "available": false
}

# Delete Slots for a Date
DELETE /api/doctor-slots/delete
Content-Type: application/json

{
  "doctorId": 1,
  "date": "2025-11-15"
}
